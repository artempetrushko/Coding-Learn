using System;
using System.Collections.Generic;
using System.Linq;

public class Level4Task3
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxCount = 52;
        var maxSize = 14;
        return Check(random, maxCount, maxSize) && Check(random, maxCount, maxSize);
    }

    private bool Check(Random random, int maxCount, int maxSize)
    {
        var array = CreateArray(random, maxCount, maxSize);
        var size = random.Next(maxSize);
        var tested= MakeMove(array, size);
        var controled = -1;
        if (size != 1)
        {
            foreach (var item in array)
            {
                if (item > size && item < controled)
                    controled = item;
            }
            if (controled == -1 && array.Contains(1))
                controled = 1;
        }
        return tested==controled;
    }

    private static int[] CreateArray(Random random, int maxCount, int maxSize)
    {
        var array = new int[random.Next(maxCount)];
        for (int i = 0; i < array.Length; i++)
        {
            array[i] = random.Next(1,maxSize);
        }

        return array;
    }
}
