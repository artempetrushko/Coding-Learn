public class Level2Task2
{
    private bool _isLogSkipped;
    private bool _isLogPut;
    private bool _isLogChopped;
    public void SkipLog()
    {
        _isLogSkipped = true;
    }
    public void PutLogInCart()
    {
        _isLogPut = true;
    }
    public void ÑhopLog()
    {
        _isLogChopped = true;
    }
    public bool isTaskCompleted()
    {
        MakeDecision(3);
        var firstCheck = _isLogSkipped && !_isLogPut && !_isLogChopped;
        _isLogSkipped = false;
        MakeDecision(5);
        var secondCheck = !_isLogSkipped && _isLogPut && !_isLogChopped;
        _isLogPut=false;
        MakeDecision(7);
        var thirdCheck = !_isLogSkipped && !_isLogPut &&_isLogChopped;
        _isLogChopped=false;
        MakeDecision(2);
        var fourthCheck = _isLogSkipped && !_isLogPut && !_isLogChopped;
        _isLogSkipped = false;
        MakeDecision(8);
        var fifthCheck = !_isLogSkipped && !_isLogPut && _isLogChopped;
        _isLogChopped = false;
        return firstCheck&&secondCheck&&thirdCheck&&fourthCheck&&fifthCheck;
    }
