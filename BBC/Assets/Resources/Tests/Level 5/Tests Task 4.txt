using System;
using System.Collections.Generic;
using System.Linq;

public class Level5Task4
{   
//<playerCode>
    public enum Direction
    {
        West, Nort, South, East
    }
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxCount = 1000;
        var maxSize = 4;
        return Check(random, maxCount, maxSize) && Check(random, maxCount, maxSize);
    }

    private bool Check(Random random, int maxCount, int maxSize)
    {
        var array = CreateArray(random, maxCount, maxSize);
        var testedArray = TranslateDirection(array.Select(x=>(Direction)x).ToArray());
        var secondCheck = true;
        for (int i = 0; i < testedArray.Length; i++)
        {
            if (testedArray[i] != array[i])
                secondCheck = false;
        }
        return secondCheck;
    }

    private static int[] CreateArray(Random random, int maxCount, int maxSize)
    {
        var array = new int[random.Next(maxCount)];
        for (int i = 0; i < array.Length; i++)
        {
            array[i] = random.Next( maxSize);
        }

        return array;
    }
    
}
