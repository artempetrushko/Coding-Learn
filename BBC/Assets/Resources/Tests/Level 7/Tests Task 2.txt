using System;
using System.Collections.Generic;

public class Level7Task2
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        return Check(random)&&Check(random);
    }

    private bool Check(Random random)
    {
        var firstValue = "firstValue";
        var list = new List<string>();
        list.Add(firstValue);
        var maxCount = 100;
        var max = 1000;
        var count = random.Next(10, maxCount);
        for (var i = 0; i < count; i++)
        {
            list.Add($"{random.Next(max)}");
        }
        RemoveFirst(list);
        return !list.Contains(firstValue)&&list.Count==count;
    }
    //<playerCode>
}
