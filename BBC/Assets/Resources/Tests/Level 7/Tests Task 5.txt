using System;
using System.Collections.Generic;
using System.Linq;

public class Level7Task1
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxCount = 1000;
        var maxSize = 10000;
        return Check(random, maxCount, maxSize)&& Check(random, maxCount, maxSize);
    }

    private bool Check(Random random, int maxCount, int maxSize)
    {
        var array = CreateArray(random, maxCount, maxSize);
        var firstCheck = array.Reverse().ToList();
        firstCheck.Reverse();
        var testedList = GetReversedList(array);
        return StringListEquals(firstCheck,testedList) ;
    }


    private bool StringListEquals(List<string> arr1, List<string> arr2)
    {
        if (arr1.Count != arr2.Count)
            return false;
        for (int i = 0; i < arr1.Count; i++)
            if (arr1[i] != arr2[i])
                return false;
        return true;
    }
    private static string[] CreateArray(Random random, int maxCount, int maxSize)
    {
        var array = new string[random.Next(maxCount)];
        for (int i = 0; i < array.Length; i++)
        {
            array[i] =$"{random.Next(10,maxSize)}";
        }
        return array;
    }

}
