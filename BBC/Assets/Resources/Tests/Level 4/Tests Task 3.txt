using System;
using System.Collections.Generic;


public class Level3Task3
{
//<playerCode>
    private int _totalStepsCount;
    private int _neededStepsCount;

    public bool IsPathClear() => _totalStepsCount == _neededStepsCount;
    public void UseTool() => _totalStepsCount++;

    public bool isTaskCompleted()
    {
        var maxCount = 100;
        var random = new Random();
        UpdateTestData(maxCount, random);
        ClearPath();
        var firstTest=IsPathClear();
        UpdateTestData(maxCount, random);
        ClearPath();
        var secondTest=IsPathClear();
        UpdateTestData(maxCount, random);
        ClearPath();
        var thirdTest=IsPathClear();
        UpdateTestData(maxCount, random);
        ClearPath();
        var fourthTest=IsPathClear();
        return firstTest && secondTest && thirdTest && fourthTest;
    }

    private void UpdateTestData(int maxCount, Random random)
    {
        _totalStepsCount = 0;
        _neededStepsCount = random.Next(maxCount);

    }
    
}
