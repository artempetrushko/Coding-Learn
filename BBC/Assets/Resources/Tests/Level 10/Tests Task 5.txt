using System;
using System.Collections.Generic;
using System.Linq;

public class Level4Task1
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxCount = 1000;
        var maxSize = 10000;
        return Check(random, maxCount, maxSize)&& Check(random, maxCount, maxSize);
    }

    private bool Check(Random random, int maxCount, int maxSize)
    {
        var fitstArray=CreateArray(random, maxCount, maxSize);
        var secondArray=CreateArray(random, maxCount, maxSize);
        var correct=fitstArray.Concat(secondArray).OrderByDescending(x=>x).ToArray();
        return correct.SequenceEqual(MergeSort(fitstArray,secondArray));
    }


    private static int[] CreateArray(Random random, int maxCount, int maxSize)
    {
        var array = new int[random.Next(maxCount)];
        for (var i = 0; i < array.Length; i++)
        {
            array[i] =random.Next(maxSize);
        }

        return array;
    }

}
