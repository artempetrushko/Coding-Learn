using System;
using System.Text;
using System.Text.RegularExpressions;

public class Level6Task7
{
    private readonly string[] pattern= new string[]
        {
            "00000000000000000[F]",
            "00000000000000000000",
            "00000000000000000000",
            "00000000000000000000",
            "00000000000000000000",
            "00000000000000000000",
            "00000000000000000000",
            "[S]00000000000000000"
        };
    public bool isTaskCompleted()
    {
        var random = new Random();
        return Check(random)&& Check(random);
    }

    private bool Check(Random random)
    {
        var array = CreateArray(random);
        return StringArrEquals(CommonCheck(array), FixPathMatrix(array));
    }
    private bool StringArrEquals(string[] arr1, string[] arr2)
    {
        if (arr1.Length != arr2.Length)
            return false;
        for (int i = 0; i < arr1.Length; i++)
            if (arr1[i] != arr2[i])
                return false;
        return true;
    }
    string[] CommonCheck(string[] arr)
    {
        var rgx = new Regex("[23456789]");
        var taskCorrectArray2 = new string[arr.Length];
        for (int i = 0; i < arr.Length; i++)
        {
            taskCorrectArray2[i] = rgx.Replace(arr[i], "0");
        }
        return taskCorrectArray2;
    }
    private string[] CreateArray(Random random)
    {
        var array = new string[pattern.Length];
        for (int i = 0; i < array.Length; i++)
        {
            var sb = new StringBuilder();
            for (int j = 0; j < pattern[0].Length; j++)
            {
                if(pattern[i][j] == '0')
                    sb.Append((char)random.Next(48, 57));
                else sb.Append(pattern[i][j]);
            }
            array[i] = sb.ToString();
        }

        return array;
    }

}
