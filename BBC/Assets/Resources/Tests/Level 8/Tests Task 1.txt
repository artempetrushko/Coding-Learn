using System;
using System.Linq;
using System.Collections.Generic;

public class Level5Task1
{
    double Pow(double value, int pow)
    {
        if (pow == 0)
            return 1;
        if (pow > 0)
            return Pow(value, pow - 1) * value;
        return 1.0 / Pow(value, -pow);
    }
    public bool isTaskCompleted()
    {
        var random=new Random();
        var value=random.NextDouble()+random.Next(10);
        var pow=random.Next(-20,20);
        var result=Pow(value, pow);;
        var finalresult=result.ToString().Split('.')[1].Take(4).Select(x=>int.Parse($"{x}")).ToList();
        var testedResult = PowAndTake(value, pow);
        return IntListEquals(finalresult,testedResult);
    }


    private bool IntListEquals(List<int> arr1, List<int> arr2)
    {
        if (arr1.Count != arr2.Count)
            return false;
        for (int i = 0; i < arr1.Count; i++)
            if (arr1[i] != arr2[i])
                return false;
        return true;
    }
}
