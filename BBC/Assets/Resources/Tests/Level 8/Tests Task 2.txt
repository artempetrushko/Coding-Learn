using System;
using System.Collections.Generic;
using System.Numerics;

public class Level8Task2
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxSize = 15;
        return Check(random, maxSize)&& Check(random, maxSize);
    }

    private bool Check(Random random, int maxSize)
    {
        var n=random.Next(maxSize);
        var result = Fact(n);
        var testedResult = Factorial(n);
        return result==testedResult;
    }


    private static string CreateArray(Random random,int maxSize)
    {
        var str = new string('1',random.Next(maxSize));
        return str;
    }
    public static BigInteger Fact(BigInteger n)
    {
        if (n == 0)
            return 1;
        else
            return n * Fact(n - 1);
    }

}
