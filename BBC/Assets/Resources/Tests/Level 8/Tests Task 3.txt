using System;
using System.Collections.Generic;
using System.Numerics;

public class Level5Task3
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxSize = 20;
        return Check(random, maxSize)&& Check(random, maxSize);
    }

    private bool Check(Random random, int maxSize)
    {
        var n = random.Next(maxSize);
        var fib=GetFib(n);
        var testedFib = Fibonacci(n);
        return fib==testedFib;
    }


    private static BigInteger GetFib(BigInteger n)
    {
        return n > 1 ? GetFib(n - 1) + GetFib(n - 2) : n;
    }

}
