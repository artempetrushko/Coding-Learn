using System;
using System.Collections.Generic;

public class Level8Task4
{
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxSize = 10000;
        return Check(random, maxSize)&& Check(random, maxSize)&& Check(random, maxSize)&& Check(random, maxSize)&& Check(random, maxSize)&& Check(random, maxSize)&& Check(random, maxSize);
    }

    private bool Check(Random random, int maxSize)
    {
        var number=(uint)random.Next(maxSize);

        return IsPrimeCorrect(number)==IsPrime(number);
    }



    static bool IsPrimeCorrect(uint n, uint k = 2)
    {
        if (k * k > n) return true;
        if (n % k == 0) return false;
        return IsPrimeCorrect(n, k + 1);
    }

}
