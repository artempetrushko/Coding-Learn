using System;
using System.Linq;
using System.Collections.Generic;

public class Level8Task1
{
//<playerCode>
    public bool isTaskCompleted()
    {
        var random = new Random();
        var maxSize = 25;
        return Check(random, maxSize)&& Check(random, maxSize);
    }

    private bool Check(Random random, int maxSize)
    {
        var n = random.Next(maxSize);
        List<int> list = GetSequence(n);
        return list.GroupBy(x => x).Where(x => x.Key != x.Count()).Count() == 0;

    }
    
}
