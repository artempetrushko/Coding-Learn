using System;
using System.Collections.Generic;


public class Level1Task5
{
//<playerCode>   
    public readonly Log Log = new Log();
    public void TakeLog(object log)
    {
        if(Log.Equals(log))
            ((Log)log).TakeLog();
    }
    public void MoveLog(object log)
    {
        if (Log.Equals(log))
            ((Log)log).MoveLog();
    }
    public void PutLog(object log)
    {
        if (Log.Equals(log))
            ((Log)log).PutLog();
    }
    public bool isTaskCompleted()
    {
        RemoveLog(Log);
        return Log.IsRemoved();
    }
    
}
public class Log
{
    private bool _isLogTaken;
    private bool _isLogMoved;
    private bool _isLogPut;
    public void TakeLog()
    {
        if (!_isLogTaken && !_isLogMoved && !_isLogPut)
        {
            _isLogTaken = true;
        }
    }
    public void MoveLog()
    {
        if (_isLogTaken && !_isLogMoved && !_isLogPut)
        {
            _isLogMoved = true;
        }
    }
    public void PutLog()
    {
        if (_isLogTaken && _isLogMoved && !_isLogPut)
        {
            _isLogPut = true;
        }
    }
    public bool IsRemoved() => _isLogTaken && _isLogMoved && _isLogPut;
}
